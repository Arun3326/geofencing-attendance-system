from flask import Flask, request, redirect, url_for, session, render_template import sqlite3 import bcrypt from datetime import datetime import pytz app = Flask(__name__) app.secret_key = 'your_secret_key' # Initialize the database with users and attendance tables def init_db(): conn = sqlite3.connect('attendance.db') c = conn.cursor() # Users table (with role) c.execute('''CREATE TABLE IF NOT EXISTS users ( id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT UNIQUE NOT NULL, password TEXT NOT NULL, role TEXT NOT NULL )''') # Attendance table c.execute('''CREATE TABLE IF NOT EXISTS attendance ( id INTEGER PRIMARY KEY AUTOINCREMENT, username TEXT NOT NULL, timestamp DATETIME DEFAULT CURRENT_TIMESTAMP )''') conn.commit() conn.close() init_db() # Login route @app.route('/', methods=['GET', 'POST']) def login(): if request.method == 'POST': username = request.form['username'] password = request.form['password'] conn = sqlite3.connect('attendance.db') c = conn.cursor() c.execute("SELECT password, role FROM users WHERE username = ?", (username,)) user = c.fetchone() conn.close() if user and bcrypt.checkpw(password.encode('utf-8'), user[0]): session['username'] = username session['role'] = user[1] return redirect(url_for('mark_attendance')) else: return "Invalid username or password" return render_template('login.html') # Attendance marking route @app.route('/mark_attendance') def mark_attendance(): if 'username' not in session: return redirect(url_for('login')) username = session['username'] local_timezone = pytz.timezone('Asia/Kolkata') local_time = datetime.now(local_timezone).strftime('%Y-%m-%d %H:%M:%S') conn = sqlite3.connect('attendance.db') c = conn.cursor() c.execute("INSERT INTO attendance (username, timestamp) VALUES (?, ?)", (username, local_time)) conn.commit() conn.close() return redirect(url_for('attendance_history')) # Show history @app.route('/attendance_history') def attendance_history(): if 'username' not in session: return redirect(url_for('login')) username = session['username'] conn = sqlite3.connect('attendance.db') c = conn.cursor() c.execute("SELECT timestamp FROM attendance WHERE username=?", (username,)) records = c.fetchall() conn.close() return render_template('attendance_history.html', username=username, records=records) # Optional map view route (keep if you have map.html) @app.route('/map') def map_view(): if 'username' not in session: return redirect(url_for('login')) return render_template("map.html") # Logout route @app.route('/logout') def logout(): session.clear() return redirect(url_for('login')) if __name__ == '__main__': app.run(debug=True)
